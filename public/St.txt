# Структура проекта CryptoLogoWall

```
/
├── index.php                  # Главная страница
├── config.php                 # Конфигурация подключения к БД и другие настройки
├── .htaccess                  # Настройки Apache для безопасности и ЧПУ
├── robots.txt                 # Настройки для поисковых роботов
├── /admin/                    # Директория админ-панели
│   ├── index.php              # Главная страница админ-панели
│   ├── logos.php              # Управление логотипами
│   ├── reviews.php            # Управление отзывами
│   ├── users.php              # Управление пользователями
│   ├── translations.php       # Управление переводами
│   └── logout.php             # Выход из админ-панели
├── /assets/                   # Ресурсы сайта
│   ├── /css/                  # Стили CSS
│   │   └── tailwind.css       # Базовый CSS Tailwind
│   ├── /js/                   # JavaScript файлы
│   │   ├── three.min.js       # Библиотека Three.js
│   │   ├── main.js            # Основной JS файл
│   │   ├── wall.js            # JS для стены логотипов
│   │   └── admin.js           # JS для админ-панели
│   ├── /img/                  # Изображения сайта
│   └── /fonts/                # Шрифты
├── /uploads/                  # Загруженные логотипы
├── /includes/                 # PHP-компоненты для подключения
│   ├── db.php                 # Класс для работы с БД
│   ├── auth.php               # Аутентификация и авторизация
│   ├── lang.php               # Система мультиязычности
│   ├── security.php           # Функции безопасности
│   ├── telegramBot.php        # Интеграция с Telegram Bot
│   ├── header.php             # Шапка сайта
│   └── footer.php             # Подвал сайта
├── /lang/                     # Языковые файлы
│   ├── en.php                 # Английский язык
│   ├── ru.php                 # Русский язык
│   └── uk.php                 # Украинский язык
└── /sql/                      # SQL файлы для создания БД
    └── cryptologowall.sql     # Структура БД
```
